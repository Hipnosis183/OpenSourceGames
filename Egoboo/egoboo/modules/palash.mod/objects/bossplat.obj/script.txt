// This reads an inscription, if a G'nome is standing on it
IfTimeOut
  tmpargument = rand & 7 + 10
  SetTime

  // Check the plate passages
  tmpx = 0 // Can't read message
  tmpdistance = 0 // Set to 1 if a message is to be displayed
  tmpturn = 0

  tmpargument = 31
  SetTargetToWhoeverIsInPassage
    IfTargetIsAPlayer
      tmpdistance = 1
      tmpargument = [GNOM]
      IfTargetHasID
        tmpx = 2 // Message
        
      //Check for truesight as well
      IfTargetCanSeeInvisible
        IfTargetCanSeeKurses
          tmpx = 2 // Message

  tmpargument = 32
  SetTargetToWhoeverIsInPassage
    IfTargetIsAPlayer
      tmpdistance = 1
      tmpargument = [GNOM]
      IfTargetHasID
        tmpx = 3 // Message
        
      //Check for truesight as well
      IfTargetCanSeeInvisible
        IfTargetCanSeeKurses
          tmpx = 3 // Message

  tmpargument = 33
  SetTargetToWhoeverIsInPassage
    IfTargetIsAPlayer
      tmpdistance = 1
      tmpargument = [GNOM]
      IfTargetHasID
        tmpx = 4 // Message
        
      //Check for truesight as well
      IfTargetCanSeeInvisible
        IfTargetCanSeeKurses
          tmpx = 4 // Message

  tmpargument = 34
  SetTargetToWhoeverIsInPassage
    IfTargetIsAPlayer
      tmpdistance = 1
      tmpargument = [GNOM]
      IfTargetHasID
        tmpx = 5 // Message
        
      //Check for truesight as well
      IfTargetCanSeeInvisible
        IfTargetCanSeeKurses
          tmpx = 5 // Message

  IfDistanceIsMoreThanTurn
    // Display the message
    tmpargument = 150
    SetTime

    tmpy = 0
    IfXIsMoreThanY
      tmpargument = 1 // "The inscription reads..."
      SendMessage
    tmpargument = tmpx // The message
    SendMessage
  Else
    // No messages
    tmpargument = 10
    SetTime

End					// All done
