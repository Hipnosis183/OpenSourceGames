    tmpdistance = 11025
    PlaySound

  KeepAction				  //
  MakeCrushValid
IfCrushed

  tmpy = selfy
  tmpdistance = selfz
  tmpargument = 4
  SpawnExactParticle
  SpawnExactParticle
  SpawnExactParticle
  SpawnExactParticle
  GoPoof

  SendMessageNear
IfDropped				// Make it lie on floor
  KeepAction				  //
IfHitGround				// Play the drop sound

  tmpdistance = rand % 2047 + 7000	  //
  PlaySound				  //
IfGrabbed				// Tell them what they've won...

    tmpargument = 0			  //
